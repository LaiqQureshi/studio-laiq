                ${imageUrl ? `<img src="${imageUrl}" alt="${prompt.title}" onerror="this.parentElement.innerHTML='<div style=\'display:flex;align-items:center;justify-content:center;height:100%;color:#888;\'>🎨</div>'">` : '<div style="display:flex;align-items:center;justify-content:center;height:100%;color:#888;">🎨</div>'}

            </div>

            <div class="card-content">

                <div class="card-title">${prompt.title}</div>

                <div class="card-style">${prompt.style}</div>

                <div class="card-preview">${prompt.preview.substring(0, 120)}...</div>

                <button class="btn-copy" onclick="showPrompt(${index})">📋 View & Copy Prompt</button>

            </div>

        `;

        grid.appendChild(card);

    });


    let currentPromptIndex = 0;

    

    function showPrompt(index) {

        currentPromptIndex = index;

        document.getElementById('modalTitle').innerText = prompts[index].title;

        document.getElementById('modalPrompt').innerText = prompts[index].full;

        document.getElementById('promptModal').style.display = 'flex';

    }


    function closeModal() {

        document.getElementById('promptModal').style.display = 'none';

    }


    function copyModalPrompt() {

        const promptText = prompts[currentPromptIndex].full;

        navigator.clipboard.writeText(promptText).then(() => {

            alert('✅ Prompt copied to clipboard!');

        });

    }


    window.onclick = function(event) {

        const modal = document.getElementById('promptModal');

        if (event.target === modal) {

            closeModal();

        }

    }
